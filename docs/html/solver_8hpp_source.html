141 template<typename C>
142 void maze::solver::BFSandDFSBase<C>::solve(Maze &maze)
143 {
144  // Reached the source vertex.
145  if (maze.solved && currPaint == maze.source()) {
146  maze.painted = true;
147  }
148  // If maze is solved and solution path in maze is still unpainted, paint it.
149  if (maze.solved && !maze.painted) {
150  currPaint = details::PaintSolutionPath(maze, parent, currPaint);
151  }
152  // Add source vertex to the stack/queue.
153  else if (!maze.source()->visited) {
154  parent[maze.source()] = nullptr;
155  container.push(maze.source());
156  maze.source()->visited = true;
157  }
158  // Exploring maze.
159  else if (!container.empty() && !pathFound) {
160  auto top = extractNext();
161  top->visited = true;
162 
163  // If reached the destination, exit.
164  if (top == maze.destination()) {
165  pathFound = true;
166  return;
167  }
168 
169  // Add unvisited neighbors of top to the stack/queue.
170  for (auto &n : details::AccessibleUnvisitedNeighbors(top, maze)) {
171  parent[n] = top;
172  container.push(n);
173  }
174  }
175  // If path is found, start painting maze.
176  else if (pathFound) {
177  currPaint = maze.destination();
178  maze.solved = true;
179  }
180  // There is no path from maze.source() to maze.destination().
181  else if (!container.empty() && !maze.solved) {
182  throw PathNotFoundException{};
183  }
184 }
